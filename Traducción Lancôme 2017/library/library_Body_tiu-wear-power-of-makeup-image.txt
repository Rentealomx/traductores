$( document ).ready(function() {

  //select your face
  $(".beforeandaftercontainer:first").addClass('active');
  $(".videobeforeandafter:first").addClass('active');  
  $(".videoright div:first img:first").addClass('active');  
  $(".videoright div:first img:eq(1)").addClass('hidden');  

  //container_width = $(".beforeandaftercontainer").width();
  //$(".beforeandaftercontainer img").width(container_width);

  windowsize = $(window).width();
  if(windowsize<1024) {
    $(".beforeandaftercontainer img").width(windowsize-50);
    $(".beforeandaftercontainer img").height('auto');
    newheight = $(".beforeandaftercontainer img").height();
    $(".beforeandaftercontainer img").height(newheight);
    $("#youtubeVideo video").width(windowsize-50);
    $("#youtubeVideo video").height(newheight);

    $("#mainheadingselect").html($("#selectyourshape").html());
    $("#mainheading").hide();
    $("#selectyourshape").html('');
    
  }

  
  $( ".selectshape img" ).on( "click", function() {
    selectedContainer = $(this).data('beforeandafter');    
    $(".beforeandaftercontainer").removeClass('active');
    $("#"+selectedContainer).addClass('active');
    $(".videobeforeandafter").removeClass('active');
    $("#video"+selectedContainer).addClass('active');    
    $("#video"+selectedContainer+" img:first").removeClass('hidden').addClass('active');    
    $("#video"+selectedContainer+" img:eq(1)").removeClass('active');        
    $("#youtubeVideo").removeClass('active');
    $("#youtubeVideo div").removeClass('active');
    $(".rigthCol .topheading").html('Watch how to');
    $("#video"+selectedContainer+" img.playbutton").removeClass('hidden').addClass('active');
  });
  $( ".videobeforeandafter img" ).on( "click", function() {
        selectedContainer = $(this).data('beforeandafter');
        $(".beforeandaftercontainer").removeClass('active');
        $("#youtubeVideo div").removeClass('active');
        $("#videoc"+selectedContainer).addClass('active');  
        if($("#youtubeVideo").hasClass('active')) {
        $("#video"+selectedContainer+" img:first").removeClass('hidden').addClass('active');  
        $("#video"+selectedContainer+" img:eq(1)").removeClass('active').addClass('hidden');      
        $("#youtubeVideo").removeClass('active');
        $("#youtubeVideo div").removeClass('active');    
        $("#"+selectedContainer).addClass('active');
        $(".rigthCol .topheading").html('Watch how to');
        $("#video"+selectedContainer+" img.playbutton").removeClass('hidden').addClass('active');
    }
    else {
        $("#video"+selectedContainer+" img:first").removeClass('active').addClass('hidden');  
        $("#video"+selectedContainer+" img:eq(1)").removeClass('hidden').addClass('active');      
        $("#youtubeVideo").addClass('active');
        $(".rigthCol .topheading").html('View before/after');
        $("#video"+selectedContainer+" img.playbutton").removeClass('active').addClass('hidden');
    }
    
  });
  
  $('#beforeandafter').beforeAfter({
        showFullLinks : false,
        baseUrl : $('#mainheadingselect').attr('data-base-url').toString().replace(/https?:\/\/.+?\/.+?\/on\/demandware.static\//, "/on/demandware.static/")
  });
  $('#beforeandafter2').beforeAfter({
        showFullLinks : false,
        baseUrl : $('#mainheadingselect').attr('data-base-url').toString().replace(/https?:\/\/.+?\/.+?\/on\/demandware.static\//, "/on/demandware.static/")
  });

        showFullLinks : false,

  });

        showFullLinks : false,

    });

});
